106/**
107 * A stacked area renderer for the {@link XYPlot} class.
108 * <br><br>
109 * The example shown here is generated by the
110 * <code>StackedXYAreaRendererDemo1.java</code> program included in the
111 * JFreeChart demo collection:
112 * <br><br>
113 * <img src="../../../../../images/StackedXYAreaRendererSample.png"
114 * alt="StackedXYAreaRendererSample.png" />
115 * <br><br>
116 * SPECIAL NOTE:  This renderer does not currently handle negative data values
117 * correctly.  This should get fixed at some point, but the current workaround
118 * is to use the {@link StackedXYAreaRenderer2} class instead.
119 */
120public class StackedXYAreaRenderer extends XYAreaRenderer
121        implements Cloneable, PublicCloneable, Serializable {

317     * Initialises the renderer. This method will be called before the first
318     * item is rendered, giving the renderer an opportunity to initialise any
319     * state information it wants to maintain.
320     *
321     * @param g2  the graphics device.
322     * @param dataArea  the area inside the axes.
323     * @param plot  the plot.
324     * @param data  the data.
325     * @param info  an optional info collection object to return data back to
326     *              the caller.
327     *
328     * @return A state object that should be passed to subsequent calls to the
329     *         drawItem() method.
330     */
331    public XYItemRendererState initialise(Graphics2D g2,
332                                          Rectangle2D dataArea,
333                                          XYPlot plot,
334                                          XYDataset data,
335                                          PlotRenderingInfo info) {
336
337        XYItemRendererState state = new StackedXYAreaRendererState(info);
338        // in the rendering process, there is special handling for item
339        // zero, so we can't support processing of visible data items only
340        state.setProcessVisibleItemsOnly(false);
341        return state;
342    }

375    /**
376     * Draws the visual representation of a single data item.
377     *
378     * @param g2  the graphics device.
379     * @param state  the renderer state.
380     * @param dataArea  the area within which the data is being drawn.
381     * @param plot  the plot (can be used to obtain standard color information
382     *              etc).
383     * @param domainAxis  the domain axis.
384     * @param rangeAxis  the range axis.
385     * @param dataset  the dataset.
386     * @param series  the series index (zero-based).
387     * @param item  the item index (zero-based).
388     * @param pass  the pass index.
389     *
390     * @throws ClassCastException if <code>state</code> is not an instance of
391     *         <code>StackedXYAreaRendererState</code> or <code>dataset</code>
392     *         is not an instance of {@link TableXYDataset}.
393     */
394    public void drawItem(Graphics2D g2, XYItemRendererState state,
395            Rectangle2D dataArea, XYPlot plot, ValueAxis domainAxis,
396            ValueAxis rangeAxis, XYDataset dataset, int series, int item,
397            boolean selected, int pass) {
398
399        PlotOrientation orientation = plot.getOrientation();
400        StackedXYAreaRendererState areaState
401            = (StackedXYAreaRendererState) state;
402        // Get the item count for the series, so that we can know which is the
403        // end of the series.
404        TableXYDataset tdataset = (TableXYDataset) dataset;
405        int itemCount = tdataset.getItemCount();
406
407        // get the data point...
408        double x1 = dataset.getXValue(series, item);
409        double y1 = dataset.getYValue(series, item);
410        boolean nullPoint = false;
411        if (Double.isNaN(y1)) {
412            y1 = 0.0;
413            nullPoint = true;
414        }
415
416        //  Get height adjustment based on stack and translate to Java2D values
417        double ph1 = getPreviousHeight(tdataset, series, item);
418        double transX1 = domainAxis.valueToJava2D(x1, dataArea,
419                plot.getDomainAxisEdge());
420        double transY1 = rangeAxis.valueToJava2D(y1 + ph1, dataArea,
421                plot.getRangeAxisEdge());
422
423        Paint seriesPaint = getItemPaint(series, item, selected);
424        Stroke seriesStroke = getItemStroke(series, item, selected);
425
426        if (pass == 0) {
427            //  On first pass render the areas, line and outlines
428
429            if (item == 0) {
430                // Create a new Area for the series
431                areaState.setSeriesArea(new Polygon());
432                areaState.setLastSeriesPoints(
433                        areaState.getCurrentSeriesPoints());
434                areaState.setCurrentSeriesPoints(new Stack());
435
436                // start from previous height (ph1)
437                double transY2 = rangeAxis.valueToJava2D(ph1, dataArea,
438                        plot.getRangeAxisEdge());
439
440                // The first point is (x, 0)
441                if (orientation == PlotOrientation.VERTICAL) {
442                    areaState.getSeriesArea().addPoint((int) transX1,
443                            (int) transY2);
444                }
445                else if (orientation == PlotOrientation.HORIZONTAL) {
446                    areaState.getSeriesArea().addPoint((int) transY2,
447                            (int) transX1);
448                }
449            }
450
451            // Add each point to Area (x, y)
452            if (orientation == PlotOrientation.VERTICAL) {
453                Point point = new Point((int) transX1, (int) transY1);
454                areaState.getSeriesArea().addPoint((int) point.getX(),
455                        (int) point.getY());
456                areaState.getCurrentSeriesPoints().push(point);
457            }
458            else if (orientation == PlotOrientation.HORIZONTAL) {
459                areaState.getSeriesArea().addPoint((int) transY1,
460                        (int) transX1);
461            }
462
463            if (getPlotLines()) {
464                if (item > 0) {
465                    // get the previous data point...
466                    double x0 = dataset.getXValue(series, item - 1);
467                    double y0 = dataset.getYValue(series, item - 1);
468                    double ph0 = getPreviousHeight(tdataset, series, item - 1);
469                    double transX0 = domainAxis.valueToJava2D(x0, dataArea,
470                            plot.getDomainAxisEdge());
471                    double transY0 = rangeAxis.valueToJava2D(y0 + ph0,
472                            dataArea, plot.getRangeAxisEdge());
473
474                    if (orientation == PlotOrientation.VERTICAL) {
475                        areaState.getLine().setLine(transX0, transY0, transX1,
476                                transY1);
477                    }
478                    else if (orientation == PlotOrientation.HORIZONTAL) {
479                        areaState.getLine().setLine(transY0, transX0, transY1,
480                                transX1);
481                    }
482                    g2.draw(areaState.getLine());
483                }
484            }
485
486            // Check if the item is the last item for the series and number of
487            // items > 0.  We can't draw an area for a single point.
488            if (getPlotArea() && item > 0 && item == (itemCount - 1)) {
489
490                double transY2 = rangeAxis.valueToJava2D(ph1, dataArea,
491                        plot.getRangeAxisEdge());
492
493                if (orientation == PlotOrientation.VERTICAL) {
494                    // Add the last point (x,0)
495                    areaState.getSeriesArea().addPoint((int) transX1,
496                            (int) transY2);
497                }
498                else if (orientation == PlotOrientation.HORIZONTAL) {
499                    // Add the last point (x,0)
500                    areaState.getSeriesArea().addPoint((int) transY2,
501                            (int) transX1);
502                }
503
504                // Add points from last series to complete the base of the
505                // polygon
506                if (series != 0) {
507                    Stack points = areaState.getLastSeriesPoints();
508                    while (!points.empty()) {
509                        Point point = (Point) points.pop();
510                        areaState.getSeriesArea().addPoint((int) point.getX(),
511                                (int) point.getY());
512                    }
513                }
514
515                //  Fill the polygon
516                g2.setPaint(seriesPaint);
517                g2.setStroke(seriesStroke);
518                g2.fill(areaState.getSeriesArea());
519
520                //  Draw an outline around the Area.
521                if (isOutline()) {
522                    g2.setStroke(lookupSeriesOutlineStroke(series));
523                    g2.setPaint(lookupSeriesOutlinePaint(series));
524                    g2.draw(areaState.getSeriesArea());
525                }
526            }
527
528            int domainAxisIndex = plot.getDomainAxisIndex(domainAxis);
529            int rangeAxisIndex = plot.getRangeAxisIndex(rangeAxis);
530            XYCrosshairState crosshairState = state.getCrosshairState();
531            updateCrosshairValues(crosshairState, x1, ph1 + y1, 
532                    domainAxisIndex, rangeAxisIndex, transX1, transY1,
533                    orientation);
534
535        }
536        else if (pass == 1) {
537            // On second pass render shapes and collect entity and tooltip
538            // information
539
540            Shape shape = null;
541            if (getPlotShapes()) {
542                shape = getItemShape(series, item, selected);
543                if (plot.getOrientation() == PlotOrientation.VERTICAL) {
544                    shape = ShapeUtilities.createTranslatedShape(shape,
545                            transX1, transY1);
546                }
547                else if (plot.getOrientation() == PlotOrientation.HORIZONTAL) {
548                    shape = ShapeUtilities.createTranslatedShape(shape,
549                            transY1, transX1);
550                }
551                if (!nullPoint) {
552                    if (getShapePaint() != null) {
553                        g2.setPaint(getShapePaint());
554                    }
555                    else {
556                        g2.setPaint(seriesPaint);
557                    }
558                    if (getShapeStroke() != null) {
559                        g2.setStroke(getShapeStroke());
560                    }
561                    else {
562                        g2.setStroke(seriesStroke);
563                    }
564                    g2.draw(shape);
565                }
566            }
567            else {
568                if (plot.getOrientation() == PlotOrientation.VERTICAL) {
569                    shape = new Rectangle2D.Double(transX1 - 3, transY1 - 3,
570                            6.0, 6.0);
571                }
572                else if (plot.getOrientation() == PlotOrientation.HORIZONTAL) {
573                    shape = new Rectangle2D.Double(transY1 - 3, transX1 - 3,
574                            6.0, 6.0);
575                }
576            }
577            EntityCollection entities = state.getEntityCollection();
578            if (entities != null) {
579                addEntity(entities, shape, dataset, series, item, selected,
580                        0.0, 0.0);
581            }
582        }
583    }

585    /**
586     * Calculates the stacked value of the all series up to, but not including
587     * <code>series</code> for the specified item. It returns 0.0 if
588     * <code>series</code> is the first series, i.e. 0.
589     *
590     * @param dataset  the dataset.
591     * @param series  the series.
592     * @param index  the index.
593     *
594     * @return The cumulative value for all series' values up to but excluding
595     *         <code>series</code> for <code>index</code>.
596     */
597    protected double getPreviousHeight(TableXYDataset dataset,
598                                       int series, int index) {
599        double result = 0.0;
600        for (int i = 0; i < series; i++) {
601            double value = dataset.getYValue(i, index);
602            if (!Double.isNaN(value)) {
603                result += value;
604            }
605        }
606        return result;
607    }

609    /**
610     * Tests the renderer for equality with an arbitrary object.
611     *
612     * @param obj  the object (<code>null</code> permitted).
613     *
614     * @return A boolean.
615     */
616    public boolean equals(Object obj) {
617        if (obj == this) {
618            return true;
619        }
620        if (!(obj instanceof StackedXYAreaRenderer) || !super.equals(obj)) {
621            return false;
622        }
623        StackedXYAreaRenderer that = (StackedXYAreaRenderer) obj;
624        if (!PaintUtilities.equal(this.shapePaint, that.shapePaint)) {
625            return false;
626        }
627        if (!ObjectUtilities.equal(this.shapeStroke, that.shapeStroke)) {
628            return false;
629        }
630        return true;
631    }
