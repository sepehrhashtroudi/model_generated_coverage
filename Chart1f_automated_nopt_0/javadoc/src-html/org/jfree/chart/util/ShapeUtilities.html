089     * Returns a clone of the specified shape, or <code>null</code>.  At the
090     * current time, this method supports cloning for instances of
091     * <code>Line2D</code>, <code>RectangularShape</code>, <code>Area</code>
092     * and <code>GeneralPath</code>.
093     * <p>
094     * <code>RectangularShape</code> includes <code>Arc2D</code>,
095     * <code>Ellipse2D</code>, <code>Rectangle2D</code>,
096     * <code>RoundRectangle2D</code>.
097     *
098     * @param shape  the shape to clone (<code>null</code> permitted,
099     *               returns <code>null</code>).
100     *
101     * @return A clone or <code>null</code>.
102     */
103    public static Shape clone(Shape shape) {
104        if (shape instanceof Cloneable) {
105            try {
106                return (Shape) ObjectUtilities.clone(shape);
107            }
108            catch (CloneNotSupportedException cnse) {
109            }
110        }
111        Shape result = null;
112        return result;
113    }

115    /**
116     * Tests two shapes for equality.  If both shapes are <code>null</code>,
117     * this method will return <code>true</code>.
118     * <p>
119     * In the current implementation, the following shapes are supported:
120     * <code>Ellipse2D</code>, <code>Line2D</code> and <code>Rectangle2D</code>
121     * (implicit).
122     *
123     * @param s1  the first shape (<code>null</code> permitted).
124     * @param s2  the second shape (<code>null</code> permitted).
125     *
126     * @return A boolean.
127     */
128    public static boolean equal(Shape s1, Shape s2) {
129        if (s1 instanceof Line2D && s2 instanceof Line2D) {
130            return equal((Line2D) s1, (Line2D) s2);
131        }
132        else if (s1 instanceof Ellipse2D && s2 instanceof Ellipse2D) {
133            return equal((Ellipse2D) s1, (Ellipse2D) s2);
134        }
135        else if (s1 instanceof Arc2D && s2 instanceof Arc2D) {
136            return equal((Arc2D) s1, (Arc2D) s2);
137        }
138        else if (s1 instanceof Polygon && s2 instanceof Polygon) {
139            return equal((Polygon) s1, (Polygon) s2);
140        }
141        else if (s1 instanceof GeneralPath && s2 instanceof GeneralPath) {
142            return equal((GeneralPath) s1, (GeneralPath) s2);
143        }
144        else {
145            // this will handle Rectangle2D...
146            return ObjectUtilities.equal(s1, s2);
147        }
148    }

150    /**
151     * Compares two lines are returns <code>true</code> if they are equal or
152     * both <code>null</code>.
153     *
154     * @param l1  the first line (<code>null</code> permitted).
155     * @param l2  the second line (<code>null</code> permitted).
156     *
157     * @return A boolean.
158     */
159    public static boolean equal(Line2D l1, Line2D l2) {
160        if (l1 == null) {
161            return (l2 == null);
162        }
163        if (l2 == null) {
164            return false;
165        }
166        if (!l1.getP1().equals(l2.getP1())) {
167            return false;
168        }
169        if (!l1.getP2().equals(l2.getP2())) {
170            return false;
171        }
172        return true;
173    }

197    /**
198     * Compares two arcs and returns <code>true</code> if they are equal or
199     * both <code>null</code>.
200     *
201     * @param a1  the first arc (<code>null</code> permitted).
202     * @param a2  the second arc (<code>null</code> permitted).
203     *
204     * @return A boolean.
205     */
206    public static boolean equal(Arc2D a1, Arc2D a2) {
207        if (a1 == null) {
208            return (a2 == null);
209        }
210        if (a2 == null) {
211            return false;
212        }
213        if (!a1.getFrame().equals(a2.getFrame())) {
214            return false;
215        }
216        if (a1.getAngleStart() != a2.getAngleStart()) {
217            return false;
218        }
219        if (a1.getAngleExtent() != a2.getAngleExtent()) {
220            return false;
221        }
222        if (a1.getArcType() != a2.getArcType()) {
223            return false;
224        }
225        return true;
226    }

228    /**
229     * Tests two polygons for equality.  If both are <code>null</code> this
230     * method returns <code>true</code>.
231     *
232     * @param p1  polygon 1 (<code>null</code> permitted).
233     * @param p2  polygon 2 (<code>null</code> permitted).
234     *
235     * @return A boolean.
236     */
237    public static boolean equal(Polygon p1, Polygon p2) {
238        if (p1 == null) {
239            return (p2 == null);
240        }
241        if (p2 == null) {
242            return false;
243        }
244        if (p1.npoints != p2.npoints) {
245            return false;
246        }
247        if (!Arrays.equals(p1.xpoints, p2.xpoints)) {
248            return false;
249        }
250        if (!Arrays.equals(p1.ypoints, p2.ypoints)) {
251            return false;
252        }
253        return true;
254    }

256    /**
257     * Tests two polygons for equality.  If both are <code>null</code> this
258     * method returns <code>true</code>.
259     *
260     * @param p1  path 1 (<code>null</code> permitted).
261     * @param p2  path 2 (<code>null</code> permitted).
262     *
263     * @return A boolean.
264     */
265    public static boolean equal(GeneralPath p1, GeneralPath p2) {
266        if (p1 == null) {
267            return (p2 == null);
268        }
269        if (p2 == null) {
270            return false;
271        }
272        if (p1.getWindingRule() != p2.getWindingRule()) {
273            return false;
274        }
275        PathIterator iterator1 = p1.getPathIterator(null);
276        PathIterator iterator2 = p2.getPathIterator(null);
277        double[] d1 = new double[6];
278        double[] d2 = new double[6];
279        boolean done = iterator1.isDone() && iterator2.isDone();
280        while (!done) {
281            if (iterator1.isDone() != iterator2.isDone()) {
282                return false;
283            }
284            int seg1 = iterator1.currentSegment(d1);
285            int seg2 = iterator2.currentSegment(d2);
286            if (seg1 != seg2) {
287                return false;
288            }
289            if (!Arrays.equals(d1, d2)) {
290                return false;
291            }
292            iterator1.next();
293            iterator2.next();
294            done = iterator1.isDone() && iterator2.isDone();
295        }
296        return true;
297    }

300     * Creates and returns a translated shape.
301     *
302     * @param shape  the shape (<code>null</code> not permitted).
303     * @param transX  the x translation (in Java2D space).
304     * @param transY  the y translation (in Java2D space).
305     *
306     * @return The translated shape.
307     */
308    public static Shape createTranslatedShape(Shape shape,
309                                              double transX,
310                                              double transY) {
311        if (shape == null) {
312            throw new IllegalArgumentException("Null 'shape' argument.");
313        }
314        AffineTransform transform = AffineTransform.getTranslateInstance(
315                transX, transY);
316        return transform.createTransformedShape(shape);
317    }

319    /**
320     * Translates a shape to a new location such that the anchor point
321     * (relative to the rectangular bounds of the shape) aligns with the
322     * specified (x, y) coordinate in Java2D space.
323     *
324     * @param shape  the shape (<code>null</code> not permitted).
325     * @param anchor  the anchor (<code>null</code> not permitted).
326     * @param locationX  the x-coordinate (in Java2D space).
327     * @param locationY  the y-coordinate (in Java2D space).
328     *
329     * @return A new and translated shape.
330     */
331    public static Shape createTranslatedShape(Shape shape,
332                                              RectangleAnchor anchor,
333                                              double locationX,
334                                              double locationY) {
335        if (shape == null) {
336            throw new IllegalArgumentException("Null 'shape' argument.");
337        }
338        if (anchor == null) {
339            throw new IllegalArgumentException("Null 'anchor' argument.");
340        }
341        Point2D anchorPoint = RectangleAnchor.coordinates(shape.getBounds2D(),
342                anchor);
343        AffineTransform transform = AffineTransform.getTranslateInstance(
344                locationX - anchorPoint.getX(), locationY - anchorPoint.getY());
345        return transform.createTransformedShape(shape);
346    }

350     *
351     * @param base  the shape (<code>null</code> permitted, returns
352     *              <code>null</code>).
353     * @param angle  the angle (in radians).
354     * @param x  the x coordinate for the rotation point (in Java2D space).
355     * @param y  the y coordinate for the rotation point (in Java2D space).
356     *
357     * @return the rotated shape.
358     */
359    public static Shape rotateShape(Shape base, double angle,
360                                    float x, float y) {
361        if (base == null) {
362            return null;
363        }
364        AffineTransform rotate = AffineTransform.getRotateInstance(angle, x, y);
365        Shape result = rotate.createTransformedShape(base);
366        return result;

370     * Draws a shape with the specified rotation about <code>(x, y)</code>.
371     *
372     * @param g2  the graphics device (<code>null</code> not permitted).
373     * @param shape  the shape (<code>null</code> not permitted).
374     * @param angle  the angle (in radians).
375     * @param x  the x coordinate for the rotation point.
376     * @param y  the y coordinate for the rotation point.

378    public static void drawRotatedShape(Graphics2D g2, Shape shape,
379                                        double angle, float x, float y) {
380
381        AffineTransform saved = g2.getTransform();
382        AffineTransform rotate = AffineTransform.getRotateInstance(angle, x, y);
383        g2.transform(rotate);
384        g2.draw(shape);
385        g2.setTransform(saved);

389    /** A useful constant used internally. */
390    private static final float SQRT2 = (float) Math.pow(2.0, 0.5);

392    /**
393     * Creates a diagonal cross shape.
394     *
395     * @param l  the length of each 'arm'.
396     * @param t  the thickness.
397     *
398     * @return A diagonal cross shape.
399     */
400    public static Shape createDiagonalCross(float l, float t) {
401        GeneralPath p0 = new GeneralPath();
402        p0.moveTo(-l - t, -l + t);
403        p0.lineTo(-l + t, -l - t);
404        p0.lineTo(0.0f, -t * SQRT2);
405        p0.lineTo(l - t, -l - t);
406        p0.lineTo(l + t, -l + t);
407        p0.lineTo(t * SQRT2, 0.0f);
408        p0.lineTo(l + t, l - t);
409        p0.lineTo(l - t, l + t);
410        p0.lineTo(0.0f, t * SQRT2);
411        p0.lineTo(-l + t, l + t);
412        p0.lineTo(-l - t, l - t);
413        p0.lineTo(-t * SQRT2, 0.0f);
414        p0.closePath();
415        return p0;
416    }

493    /**
494     * Creates a region surrounding a line segment by 'widening' the line
495     * segment.  A typical use for this method is the creation of a
496     * 'clickable' region for a line that is displayed on-screen.
497     *
498     * @param line  the line (<code>null</code> not permitted).
499     * @param width  the width of the region.
500     *
501     * @return A region that surrounds the line.
502     */
503    public static Shape createLineRegion(Line2D line, float width) {
504        GeneralPath result = new GeneralPath();
505        float x1 = (float) line.getX1();
506        float x2 = (float) line.getX2();
507        float y1 = (float) line.getY1();
508        float y2 = (float) line.getY2();
509        if ((x2 - x1) != 0.0) {
510            double theta = Math.atan((y2 - y1) / (x2 - x1));
511            float dx = (float) Math.sin(theta) * width;
512            float dy = (float) Math.cos(theta) * width;
513            result.moveTo(x1 - dx, y1 + dy);
514            result.lineTo(x1 + dx, y1 - dy);
515            result.lineTo(x2 + dx, y2 - dy);
516            result.lineTo(x2 - dx, y2 + dy);
517            result.closePath();
518        }
519        else {
520            // special case, vertical line
521            result.moveTo(x1 - width / 2.0f, y1);
522            result.lineTo(x1 + width / 2.0f, y1);
523            result.lineTo(x2 + width / 2.0f, y2);
524            result.lineTo(x2 - width / 2.0f, y2);
525            result.closePath();
526        }
527        return result;
528    }

533     *

546        x = Math.max(area.getMinX(), Math.min(x, area.getMaxX()));
547        y = Math.max(area.getMinY(), Math.min(y, area.getMaxY()));
548        return new Point2D.Double(x, y);

552    /**
553     * Checks, whether the given rectangle1 fully contains rectangle 2
554     * (even if rectangle 2 has a height or width of zero!).
555     *
556     * @param rect1  the first rectangle.
557     * @param rect2  the second rectangle.
558     *
559     * @return A boolean.
560     */
561    public static boolean contains(Rectangle2D rect1, Rectangle2D rect2) {
562
563        double x0 = rect1.getX();
564        double y0 = rect1.getY();
565        double x = rect2.getX();
566        double y = rect2.getY();
567        double w = rect2.getWidth();
568        double h = rect2.getHeight();
569
570        return ((x >= x0) && (y >= y0)
571                && ((x + w) <= (x0 + rect1.getWidth()))
572                && ((y + h) <= (y0 + rect1.getHeight())));
573
574    }

577    /**
578     * Checks, whether the given rectangle1 fully contains rectangle 2
579     * (even if rectangle 2 has a height or width of zero!).
580     *
581     * @param rect1  the first rectangle.
582     * @param rect2  the second rectangle.
583     *
584     * @return A boolean.
585     */
586    public static boolean intersects(Rectangle2D rect1, Rectangle2D rect2) {
587
588      double x0 = rect1.getX();
589      double y0 = rect1.getY();
590
591      double x = rect2.getX();
592      double width = rect2.getWidth();
593      double y = rect2.getY();
594      double height = rect2.getHeight();
595      return (x + width >= x0 && y + height >= y0 && x <= x0 + rect1.getWidth()
596              && y <= y0 + rect1.getHeight());
597    }

616    /**
617     * Clips the specified line to the given rectangle.
618     *
619     * @param line  the line (<code>null</code> not permitted).
620     * @param rect  the clipping rectangle (<code>null</code> not permitted).
621     *
622     * @return <code>true</code> if the clipped line is visible, and
623     *     <code>false</code> otherwise.
624     *
625     * @since 1.2.0
626     */
627    public static boolean clipLine(Line2D line, Rectangle2D rect) {
628
629        double x1 = line.getX1();
630        double y1 = line.getY1();
631        double x2 = line.getX2();
632        double y2 = line.getY2();
633
634        double minX = rect.getMinX();
635        double maxX = rect.getMaxX();
636        double minY = rect.getMinY();
637        double maxY = rect.getMaxY();
638
639        int f1 = rect.outcode(x1, y1);
640        int f2 = rect.outcode(x2, y2);
641
642        while ((f1 | f2) != 0) {
643            if ((f1 & f2) != 0) {
644                return false;
645            }
646            double dx = (x2 - x1);
647            double dy = (y2 - y1);
648            // update (x1, y1), (x2, y2) and f1 and f2 using intersections
649            // then recheck
650            if (f1 != 0) {
651                // first point is outside, so we update it against one of the
652                // four sides then continue
653                if ((f1 & Rectangle2D.OUT_LEFT) == Rectangle2D.OUT_LEFT
654                        && dx != 0.0) {
655                    y1 = y1 + (minX - x1) * dy / dx;
656                    x1 = minX;
657                }
658                else if ((f1 & Rectangle2D.OUT_RIGHT) == Rectangle2D.OUT_RIGHT
659                        && dx != 0.0) {
660                    y1 = y1 + (maxX - x1) * dy / dx;
661                    x1 = maxX;
662                }
663                else if ((f1 & Rectangle2D.OUT_BOTTOM) == Rectangle2D.OUT_BOTTOM
664                        && dy != 0.0) {
665                    x1 = x1 + (maxY - y1) * dx / dy;
666                    y1 = maxY;
667                }
668                else if ((f1 & Rectangle2D.OUT_TOP) == Rectangle2D.OUT_TOP
669                        && dy != 0.0) {
670                    x1 = x1 + (minY - y1) * dx / dy;
671                    y1 = minY;
672                }
673                f1 = rect.outcode(x1, y1);
674            }
675            else if (f2 != 0) {
676                // second point is outside, so we update it against one of the
677                // four sides then continue
678                if ((f2 & Rectangle2D.OUT_LEFT) == Rectangle2D.OUT_LEFT
679                        && dx != 0.0) {
680                    y2 = y2 + (minX - x2) * dy / dx;
681                    x2 = minX;
682                }
683                else if ((f2 & Rectangle2D.OUT_RIGHT) == Rectangle2D.OUT_RIGHT
684                        && dx != 0.0) {
685                    y2 = y2 + (maxX - x2) * dy / dx;
686                    x2 = maxX;
687                }
688                else if ((f2 & Rectangle2D.OUT_BOTTOM) == Rectangle2D.OUT_BOTTOM
689                        && dy != 0.0) {
690                    x2 = x2 + (maxY - y2) * dx / dy;
691                    y2 = maxY;
692                }
693                else if ((f2 & Rectangle2D.OUT_TOP) == Rectangle2D.OUT_TOP
694                        && dy != 0.0) {
695                    x2 = x2 + (minY - y2) * dx / dy;
696                    y2 = minY;
697                }
698                f2 = rect.outcode(x2, y2);
699            }
700        }
701
702        line.setLine(x1, y1, x2, y2);
703        return true;  // the line is visible - if it wasn't, we'd have
704                      // returned false from within the while loop above
705
706    }
