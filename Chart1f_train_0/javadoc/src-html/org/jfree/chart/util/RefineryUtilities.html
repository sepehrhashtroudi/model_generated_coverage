101    /**
102     * Computes the center point of the current screen device. If this method
103     * is called on JDK 1.4, Xinerama-aware results are returned. (See
104     * Sun-Bug-ID 4463949 for details).
105     *
106     * @return the center point of the current screen.
107     */
108    public static Point getCenterPoint ()
109    {
110      GraphicsEnvironment localGraphicsEnvironment
111              = GraphicsEnvironment.getLocalGraphicsEnvironment();
112      try {
113          Method method = GraphicsEnvironment.class.getMethod("getCenterPoint",
114                  (Class[]) null);
115          return (Point) method.invoke(localGraphicsEnvironment,
116                  (Object[]) null);
117      }
118      catch (Exception e) {
119        // ignore ... will fail if this is not a JDK 1.4 ..
120      }
121
122      Dimension s = Toolkit.getDefaultToolkit().getScreenSize();
123      return new Point (s.width / 2, s.height / 2);
124    }

126  /**
127   * Computes the maximum bounds of the current screen device. If this method
128   * is called on JDK 1.4, Xinerama-aware results are returned.
129   * (See Sun-Bug-ID 4463949 for details).
130   *
131   * @return the maximum bounds of the current screen.
132   */
133    public static Rectangle getMaximumWindowBounds() {
134      GraphicsEnvironment localGraphicsEnvironment
135              = GraphicsEnvironment.getLocalGraphicsEnvironment();
136      try {
137          Method method = GraphicsEnvironment.class.getMethod(
138                  "getMaximumWindowBounds", (Class[]) null);
139          return (Rectangle) method.invoke(localGraphicsEnvironment,
140                  (Object[]) null);
141      }
142      catch (Exception e) {
143          // ignore ... will fail if this is not a JDK 1.4 ..
144      }
145
146      Dimension s = Toolkit.getDefaultToolkit().getScreenSize();
147      return new Rectangle (0, 0, s.width, s.height);
148    }

150    /**
151     * Positions the specified frame in the middle of the screen.
152     *
153     * @param frame  the frame to be centered on the screen.
154     */
155    public static void centerFrameOnScreen(final Window frame) {
156        positionFrameOnScreen(frame, 0.5, 0.5);
157    }
158
159    /**
160     * Positions the specified frame at a relative position in the screen,
161     * where 50% is considered to be the center of the screen.
162     *
163     * @param frame  the frame.
164     * @param horizontalPercent  the relative horizontal position of the frame
165     *     (0.0 to 1.0, where 0.5 is the center of the screen).
166     * @param verticalPercent  the relative vertical position of the frame
167     *     (0.0 to 1.0, where 0.5 is the center of the screen).
168     */
169    public static void positionFrameOnScreen(Window frame,
170                                             double horizontalPercent,
171                                             double verticalPercent) {
172
173        Rectangle s = getMaximumWindowBounds();
174        Dimension f = frame.getSize();
175        int w = Math.max(s.width - f.width, 0);
176        int h = Math.max(s.height - f.height, 0);
177        int x = (int) (horizontalPercent * w) + s.x;
178        int y = (int) (verticalPercent * h) + s.y;
179        frame.setBounds(x, y, f.width, f.height);
180
181    }
182
183    /**
184     * Positions the specified frame at a random location on the screen while ensuring that the
185     * entire frame is visible (provided that the frame is smaller than the screen).
186     *
187     * @param frame  the frame.
188     */
189    public static void positionFrameRandomly(Window frame) {
190        positionFrameOnScreen(frame, Math.random(), Math.random());
191    }

193    /**
194     * Positions the specified dialog within its parent.
195     *
196     * @param dialog  the dialog to be positioned on the screen.
197     */
198    public static void centerDialogInParent(Dialog dialog) {
199        positionDialogRelativeToParent(dialog, 0.5, 0.5);
200    }
201
202    /**
203     * Positions the specified dialog at a position relative to its parent.
204     *
205     * @param dialog  the dialog to be positioned.
206     * @param horizontalPercent  the relative location.
207     * @param verticalPercent  the relative location.
208     */
209    public static void positionDialogRelativeToParent(Dialog dialog,
210                                                      double horizontalPercent,
211                                                      double verticalPercent) {
212        Dimension d = dialog.getSize();
213        Container parent = dialog.getParent();
214        Dimension p = parent.getSize();
215
216        int baseX = parent.getX() - d.width;
217        int baseY = parent.getY() - d.height;
218        int w = d.width + p.width;
219        int h = d.height + p.height;
220        int x = baseX + (int) (horizontalPercent * w);
221        int y = baseY + (int) (verticalPercent * h);
222
223        // make sure the dialog fits completely on the screen...
224        Rectangle s = getMaximumWindowBounds();
225        x = Math.min(x, (s.width - d.width));
226        x = Math.max(x, 0);
227        y = Math.min(y, (s.height - d.height));
228        y = Math.max(y, 0);
229
230        dialog.setBounds(x + s.x, y + s.y, d.width, d.height);
231
232    }

234    /**
235     * Creates a panel that contains a table based on the specified table model.
236     *
237     * @param model  the table model to use when constructing the table.
238     *
239     * @return The panel.
240     */
241    public static JPanel createTablePanel(TableModel model) {
242
243        JPanel panel = new JPanel(new BorderLayout());
244        JTable table = new JTable(model);
245        for (int columnIndex = 0; columnIndex < model.getColumnCount();
246                columnIndex++) {
247            TableColumn column = table.getColumnModel().getColumn(columnIndex);
248            Class c = model.getColumnClass(columnIndex);
249            if (c.equals(Number.class)) {
250                column.setCellRenderer(new NumberCellRenderer());
251            }
252        }
253        panel.add(new JScrollPane(table));
254        return panel;
255
256    }
