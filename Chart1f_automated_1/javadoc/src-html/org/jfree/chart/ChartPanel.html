613     * controls whether or not an offscreen <code>BufferedImage</code> is
614     * maintained for the chart.  If the buffer is used, more memory is
615     * consumed, but panel repaints will be a lot quicker in cases where the
616     * chart itself hasn't changed (for example, when another frame is moved
617     * to reveal the panel).  WARNING: If you set the <code>useBuffer</code>

620     * on JRE6 on Windows.

818     *
819     * @param chart  the chart (<code>null</code> permitted).
820     */
821    public void setChart(JFreeChart chart) {
822
823        // stop listening for changes to the existing chart
824        if (this.chart != null) {
825            this.chart.removeChangeListener(this);
826            this.chart.removeProgressListener(this);
827        }
828
829        // add the new chart
830        this.chart = chart;
831        if (chart != null) {
832            this.chart.addChangeListener(this);
833            this.chart.addProgressListener(this);
834            Plot plot = chart.getPlot();
835            this.domainZoomable = false;
836            this.rangeZoomable = false;
837            if (plot instanceof Zoomable) {
838                Zoomable z = (Zoomable) plot;
839                this.domainZoomable = z.isDomainZoomable();
840                this.rangeZoomable = z.isRangeZoomable();
841                this.orientation = z.getOrientation();
842            }
843        }
844        else {
845            this.domainZoomable = false;
846            this.rangeZoomable = false;
847        }
848        if (this.useBuffer) {
849            this.refreshBuffer = true;
850        }
851        repaint();
852
853    }

1599     * Paints the component by drawing the chart to fill the entire component,
1600     * but allowing for the insets (which will be non-zero if a border has been
1601     * set for this component).  To increase performance (at the expense of
1602     * memory), an off-screen buffer image can be used.
1603     *
1604     * @param g  the graphics device for drawing on.
1605     */
1606    public void paintComponent(Graphics g) {
1607        super.paintComponent(g);
1608        if (this.chart == null) {
1609            return;
1610        }
1611        Graphics2D g2 = (Graphics2D) g.create();
1612
1613        // first determine the size of the chart rendering area...
1614        Dimension size = getSize();
1615        Insets insets = getInsets();
1616        Rectangle2D available = new Rectangle2D.Double(insets.left, insets.top,
1617                size.getWidth() - insets.left - insets.right,
1618                size.getHeight() - insets.top - insets.bottom);
1619
1620        // work out if scaling is required...
1621        boolean scale = false;
1622        double drawWidth = available.getWidth();
1623        double drawHeight = available.getHeight();
1624        this.scaleX = 1.0;
1625        this.scaleY = 1.0;
1626
1627        if (drawWidth < this.minimumDrawWidth) {
1628            this.scaleX = drawWidth / this.minimumDrawWidth;
1629            drawWidth = this.minimumDrawWidth;
1630            scale = true;
1631        }
1632        else if (drawWidth > this.maximumDrawWidth) {
1633            this.scaleX = drawWidth / this.maximumDrawWidth;
1634            drawWidth = this.maximumDrawWidth;
1635            scale = true;
1636        }
1637
1638        if (drawHeight < this.minimumDrawHeight) {
1639            this.scaleY = drawHeight / this.minimumDrawHeight;
1640            drawHeight = this.minimumDrawHeight;
1641            scale = true;
1642        }
1643        else if (drawHeight > this.maximumDrawHeight) {
1644            this.scaleY = drawHeight / this.maximumDrawHeight;
1645            drawHeight = this.maximumDrawHeight;
1646            scale = true;
1647        }
1648
1649        Rectangle2D chartArea = new Rectangle2D.Double(0.0, 0.0, drawWidth,
1650                drawHeight);
1651
1652        // are we using the chart buffer?
1653        if (this.useBuffer) {
1654
1655            // do we need to resize the buffer?
1656            if ((this.chartBuffer == null)
1657                    || (this.chartBufferWidth != available.getWidth())
1658                    || (this.chartBufferHeight != available.getHeight())) {
1659                this.chartBufferWidth = (int) available.getWidth();
1660                this.chartBufferHeight = (int) available.getHeight();
1661                GraphicsConfiguration gc = g2.getDeviceConfiguration();
1662                this.chartBuffer = gc.createCompatibleImage(
1663                        this.chartBufferWidth, this.chartBufferHeight,
1664                        Transparency.TRANSLUCENT);
1665                this.refreshBuffer = true;
1666            }
1667
1668            // do we need to redraw the buffer?
1669            if (this.refreshBuffer) {
1670
1671                this.refreshBuffer = false; // clear the flag
1672
1673                Rectangle2D bufferArea = new Rectangle2D.Double(
1674                        0, 0, this.chartBufferWidth, this.chartBufferHeight);
1675
1676                // make the background of the buffer clear and transparent
1677                Graphics2D bufferG2 = (Graphics2D)
1678                        this.chartBuffer.getGraphics();
1679                Composite savedComposite = bufferG2.getComposite();
1680                bufferG2.setComposite(AlphaComposite.getInstance(
1681                        AlphaComposite.CLEAR, 0.0f));
1682                Rectangle r = new Rectangle(0, 0, this.chartBufferWidth,
1683                        this.chartBufferHeight);
1684                bufferG2.fill(r);
1685                bufferG2.setComposite(savedComposite);
1686
1687                if (scale) {
1688                    AffineTransform saved = bufferG2.getTransform();
1689                    AffineTransform st = AffineTransform.getScaleInstance(
1690                            this.scaleX, this.scaleY);
1691                    bufferG2.transform(st);
1692                    this.chart.draw(bufferG2, chartArea, this.anchor,
1693                            this.info);
1694                    bufferG2.setTransform(saved);
1695                }
1696                else {
1697                    this.chart.draw(bufferG2, bufferArea, this.anchor,
1698                            this.info);
1699                }
1700
1701            }
1702
1703            // zap the buffer onto the panel...
1704            g2.drawImage(this.chartBuffer, insets.left, insets.top, this);
1705
1706        }
1707
1708        // or redrawing the chart every time...
1709        else {
1710
1711            AffineTransform saved = g2.getTransform();
1712            g2.translate(insets.left, insets.top);
1713            if (scale) {
1714                AffineTransform st = AffineTransform.getScaleInstance(
1715                        this.scaleX, this.scaleY);
1716                g2.transform(st);
1717            }
1718            this.chart.draw(g2, chartArea, this.anchor, this.info);
1719            g2.setTransform(saved);
1720
1721        }
1722
1723        Iterator iterator = this.overlays.iterator();
1724        while (iterator.hasNext()) {
1725            Overlay overlay = (Overlay) iterator.next();
1726            overlay.paintOverlay(g2, this);
1727        }
1728
1729        // redraw the zoom rectangle (if present) - if useBuffer is false,
1730        // we use XOR so we can XOR the rectangle away again without redrawing
1731        // the chart
1732        drawZoomRectangle(g2, !this.useBuffer);
1733        drawSelectionShape(g2, !this.useBuffer);
1734        g2.dispose();
1735
1736        this.anchor = null;
1737
1738    }

2876     * The idea is to modify the zooming options depending on the type of chart
2877     * being displayed by the panel.
2878     *
2879     * @param x  horizontal position of the popup.
2880     * @param y  vertical position of the popup.
2881     */
2882    protected void displayPopupMenu(int x, int y) {
2883
2884        if (this.popup != null) {
2885
2886            // go through each zoom menu item and decide whether or not to
2887            // enable it...
2888            Plot plot = this.chart.getPlot();
2889            boolean isDomainZoomable = false;
2890            boolean isRangeZoomable = false;
2891            if (plot instanceof Zoomable) {
2892                Zoomable z = (Zoomable) plot;
2893                isDomainZoomable = z.isDomainZoomable();
2894                isRangeZoomable = z.isRangeZoomable();
2895            }
2896
2897            if (this.zoomInDomainMenuItem != null) {
2898                this.zoomInDomainMenuItem.setEnabled(isDomainZoomable);
2899            }
2900            if (this.zoomOutDomainMenuItem != null) {
2901                this.zoomOutDomainMenuItem.setEnabled(isDomainZoomable);
2902            }
2903            if (this.zoomResetDomainMenuItem != null) {
2904                this.zoomResetDomainMenuItem.setEnabled(isDomainZoomable);
2905            }
2906
2907            if (this.zoomInRangeMenuItem != null) {
2908                this.zoomInRangeMenuItem.setEnabled(isRangeZoomable);
2909            }
2910            if (this.zoomOutRangeMenuItem != null) {
2911                this.zoomOutRangeMenuItem.setEnabled(isRangeZoomable);
2912            }
2913
2914            if (this.zoomResetRangeMenuItem != null) {
2915                this.zoomResetRangeMenuItem.setEnabled(isRangeZoomable);
2916            }
2917
2918            if (this.zoomInBothMenuItem != null) {
2919                this.zoomInBothMenuItem.setEnabled(isDomainZoomable
2920                        && isRangeZoomable);
2921            }
2922            if (this.zoomOutBothMenuItem != null) {
2923                this.zoomOutBothMenuItem.setEnabled(isDomainZoomable
2924                        && isRangeZoomable);
2925            }
2926            if (this.zoomResetBothMenuItem != null) {
2927                this.zoomResetBothMenuItem.setEnabled(isDomainZoomable
2928                        && isRangeZoomable);
2929            }
2930
2931            this.popup.show(this, x, y);
2932        }
