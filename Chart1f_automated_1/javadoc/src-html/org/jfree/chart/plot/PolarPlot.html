197   /**

204    public PolarPlot(XYDataset dataset,
205                     ValueAxis radiusAxis,
206                     PolarItemRenderer renderer) {
207
208        super();
209
210        this.dataset = dataset;
211        if (this.dataset != null) {
212            this.dataset.addChangeListener(this);
213        }
214        this.angleTickUnit = new NumberTickUnit(DEFAULT_ANGLE_TICK_UNIT_SIZE);
215
216        this.axis = radiusAxis;
217        if (this.axis != null) {
218            this.axis.setPlot(this);
219            this.axis.addChangeListener(this);
220        }
221
222        this.renderer = renderer;
223        if (this.renderer != null) {
224            this.renderer.setPlot(this);
225            this.renderer.addChangeListener(this);
226        }
227
228        this.angleGridlinesVisible = true;
229        this.angleGridlineStroke = DEFAULT_GRIDLINE_STROKE;
230        this.angleGridlinePaint = DEFAULT_GRIDLINE_PAINT;
231
232        this.radiusGridlinesVisible = true;
233        this.radiusGridlineStroke = DEFAULT_GRIDLINE_STROKE;
234        this.radiusGridlinePaint = DEFAULT_GRIDLINE_PAINT;
235    }

342     *
343     * @see #getDataset()
344     */
345    public void setDataset(XYDataset dataset) {
346        // if there is an existing dataset, remove the plot from the list of
347        // change listeners...
348        XYDataset existing = this.dataset;
349        if (existing != null) {
350            existing.removeChangeListener(this);
351        }
352
353        // set the new dataset, and register the chart as a change listener...
354        this.dataset = dataset;
355        if (this.dataset != null) {
356            setDatasetGroup(this.dataset.getGroup());
357            this.dataset.addChangeListener(this);
358        }
359
360        // send a dataset change event to self...
361        DatasetChangeEvent event = new DatasetChangeEvent(this, this.dataset,
362                new DatasetChangeInfo());
363        // TODO: fill in real dataset change type
364        datasetChanged(event);
365    }

674    /**
675     * Generates a list of tick values for the angular tick marks.
676     *
677     * @return A list of {@link NumberTick} instances.
678     *
679     * @since 1.0.10
680     */
681    protected List refreshAngleTicks() {
682        List ticks = new ArrayList();
683        for (double currentTickVal = 0.0; currentTickVal < 360.0;
684                currentTickVal += this.angleTickUnit.getSize()) {
685            NumberTick tick = new NumberTick(new Double(currentTickVal),
686                this.angleTickUnit.valueToString(currentTickVal),
687                TextAnchor.CENTER, TextAnchor.CENTER, 0.0);
688            ticks.add(tick);
689        }
690        return ticks;
691    }

693    /**
694     * Draws the plot on a Java 2D graphics device (such as the screen or a
695     * printer).
696     * <P>
697     * This plot relies on a {@link PolarItemRenderer} to draw each
698     * item in the plot.  This allows the visual representation of the data to
699     * be changed easily.
700     * <P>
701     * The optional info argument collects information about the rendering of
702     * the plot (dimensions, tooltip information etc).  Just pass in
703     * <code>null</code> if you do not need this information.
704     *
705     * @param g2  the graphics device.
706     * @param area  the area within which the plot (including axes and
707     *              labels) should be drawn.
708     * @param anchor  the anchor point (<code>null</code> permitted).
709     * @param parentState  ignored.
710     * @param info  collects chart drawing information (<code>null</code>
711     *              permitted).
712     */
713    public void draw(Graphics2D g2,
714                     Rectangle2D area,
715                     Point2D anchor,
716                     PlotState parentState,
717                     PlotRenderingInfo info) {
718
719        // if the plot area is too small, just return...
720        boolean b1 = (area.getWidth() <= MINIMUM_WIDTH_TO_DRAW);
721        boolean b2 = (area.getHeight() <= MINIMUM_HEIGHT_TO_DRAW);
722        if (b1 || b2) {
723            return;
724        }
725
726        // record the plot area...
727        if (info != null) {
728            info.setPlotArea(area);
729        }
730
731        // adjust the drawing area for the plot insets (if any)...
732        RectangleInsets insets = getInsets();
733        insets.trim(area);
734
735        Rectangle2D dataArea = area;
736        if (info != null) {
737            info.setDataArea(dataArea);
738        }
739
740        // draw the plot background and axes...
741        drawBackground(g2, dataArea);
742        double h = Math.min(dataArea.getWidth() / 2.0,
743                dataArea.getHeight() / 2.0) - MARGIN;
744        Rectangle2D quadrant = new Rectangle2D.Double(dataArea.getCenterX(),
745                dataArea.getCenterY(), h, h);
746        AxisState state = drawAxis(g2, area, quadrant);
747        if (this.renderer != null) {
748            Shape originalClip = g2.getClip();
749            Composite originalComposite = g2.getComposite();
750
751            g2.clip(dataArea);
752            g2.setComposite(AlphaComposite.getInstance(
753                    AlphaComposite.SRC_OVER, getForegroundAlpha()));
754
755            this.angleTicks = refreshAngleTicks();
756            drawGridlines(g2, dataArea, this.angleTicks, state.getTicks());
757
758            // draw...
759            render(g2, dataArea, info);
760
761            g2.setClip(originalClip);
762            g2.setComposite(originalComposite);
763        }
764        drawOutline(g2, dataArea);
765        drawCornerTextItems(g2, dataArea);
766    }

768    /**
769     * Draws the corner text items.
770     *
771     * @param g2  the drawing surface.
772     * @param area  the area.
773     */
774    protected void drawCornerTextItems(Graphics2D g2, Rectangle2D area) {
775        if (this.cornerTextItems.isEmpty()) {
776            return;
777        }
778
779        g2.setColor(Color.black);
780        double width = 0.0;
781        double height = 0.0;
782        for (Iterator it = this.cornerTextItems.iterator(); it.hasNext();) {
783            String msg = (String) it.next();
784            FontMetrics fm = g2.getFontMetrics();
785            Rectangle2D bounds = TextUtilities.getTextBounds(msg, g2, fm);
786            width = Math.max(width, bounds.getWidth());
787            height += bounds.getHeight();
788        }
789
790        double xadj = ANNOTATION_MARGIN * 2.0;
791        double yadj = ANNOTATION_MARGIN;
792        width += xadj;
793        height += yadj;
794
795        double x = area.getMaxX() - width;
796        double y = area.getMaxY() - height;
797        g2.drawRect((int) x, (int) y, (int) width, (int) height);
798        x += ANNOTATION_MARGIN;
799        for (Iterator it = this.cornerTextItems.iterator(); it.hasNext();) {
800            String msg = (String) it.next();
801            Rectangle2D bounds = TextUtilities.getTextBounds(msg, g2,
802                    g2.getFontMetrics());
803            y += bounds.getHeight();
804            g2.drawString(msg, (int) x, (int) y);
805        }
806    }

1260    /**
1261     * Translates a (theta, radius) pair into Java2D coordinates.  If
1262     * <code>radius</code> is less than the lower bound of the axis, then
1263     * this method returns the centre point.
1264     *
1265     * @param angleDegrees  the angle in degrees.
1266     * @param radius  the radius.
1267     * @param dataArea  the data area.
1268     *
1269     * @return A point in Java2D space.
1270     */
1271    public Point translateValueThetaRadiusToJava2D(double angleDegrees,
1272                                                   double radius,
1273                                                   Rectangle2D dataArea) {
1274
1275        double radians = Math.toRadians(angleDegrees - 90.0);
1276
1277        double minx = dataArea.getMinX() + MARGIN;
1278        double maxx = dataArea.getMaxX() - MARGIN;
1279        double miny = dataArea.getMinY() + MARGIN;
1280        double maxy = dataArea.getMaxY() - MARGIN;
1281
1282        double lengthX = maxx - minx;
1283        double lengthY = maxy - miny;
1284        double length = Math.min(lengthX, lengthY);
1285
1286        double midX = minx + lengthX / 2.0;
1287        double midY = miny + lengthY / 2.0;
1288
1289        double axisMin = this.axis.getLowerBound();
1290        double axisMax =  getMaxRadius();
1291        double adjustedRadius = Math.max(radius, axisMin);
1292
1293        double xv = length / 2.0 * Math.cos(radians);
1294        double yv = length / 2.0 * Math.sin(radians);
1295
1296        float x = (float) (midX + (xv * (adjustedRadius - axisMin)
1297                / (axisMax - axisMin)));
1298        float y = (float) (midY + (yv * (adjustedRadius - axisMin)
1299                / (axisMax - axisMin)));
1300
1301        int ix = Math.round(x);
1302        int iy = Math.round(y);
1303
1304        Point p = new Point(ix, iy);
1305        return p;
1306
1307    }
