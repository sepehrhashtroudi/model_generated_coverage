041package org.jfree.data.pie;
042
043import org.jfree.data.general.*;
044import java.io.Serializable;
045import org.jfree.chart.util.PublicCloneable;
046
047/**
048 * Returns information about the selection state of items in an
049 * {@link PieDataset}.  Classes that implement this interface must also
050 * implement {@link PublicCloneable} to ensure that charts and datasets can be
051 * correctly cloned.  Likewise, classes implementing this interface must also
052 * implement {@link Serializable}.
053 * <br><br>
054 * The selection state might be part of a dataset implementation, or it could
055 * be maintained in parallel with a dataset implementation that doesn't
056 * directly support selection state.
057 *
058 * @since 1.2.0
059 */
060public interface PieDatasetSelectionState extends DatasetSelectionState {
061
062    /**
063     * Returns the number of items in the dataset.
064     *
065     * @return The number of items.
066     */
067    public int getItemCount();
068
069    /**
070     * Returns <code>true</code> if the specified item is selected, and
071     * <code>false</code> otherwise.
072     *
073     * @param key  the item key.
074     *
075     * @return A boolean.
076     */
077    public boolean isSelected(Comparable key);
078
079    /**
080     * Sets the selection state for an item in the dataset.
081     *
082     * @param key  the item key.
083     * @param selected  the selection state.
084     */
085    public void setSelected(Comparable key, boolean selected);
086
087    /**
088     * Sets the selection state for the specified item and, if requested,
089     * fires a change event.
090     *
091     * @param key  the item key.
092     * @param selected  the selection state.
093     * @param notify  notify listeners?
094     */
095    public void setSelected(Comparable key, boolean selected, boolean notify);
096
097    /**
098     * Clears all selected items.
099     */
100    public void clearSelection();
101
102    /**
103     * Send an event to registered listeners to indicate that the selection
104     * has changed.
105     */
106    public void fireSelectionEvent();
107
108}
