173    /**
174     * Tests this instance for equality with an arbitrary object.
175     *
176     * @param obj  the object (<code>null</code> permitted).
177     *
178     * @return A boolean.
179     */
180    public boolean equals(Object obj) {
181        if (obj == this) {
182            return true;
183        }
184        if (!(obj instanceof RectangleInsets)) {
185                return false;
186        }
187        final RectangleInsets that = (RectangleInsets) obj;
188        if (that.unitType != this.unitType) {
189            return false;
190        }
191        if (this.left != that.left) {
192            return false;
193        }
194        if (this.right != that.right) {
195            return false;
196        }
197        if (this.top != that.top) {
198            return false;
199        }
200        if (this.bottom != that.bottom) {
201            return false;
202        }
203        return true;
204    }

206    /**
207     * Returns a hash code for the object.
208     *
209     * @return A hash code.
210     */
211    public int hashCode() {
212        int result;
213        long temp;
214        result = (this.unitType != null ? this.unitType.hashCode() : 0);
215        temp = this.top != +0.0d ? Double.doubleToLongBits(this.top) : 0L;
216        result = 29 * result + (int) (temp ^ (temp >>> 32));
217        temp = this.bottom != +0.0d ? Double.doubleToLongBits(this.bottom) : 0L;
218        result = 29 * result + (int) (temp ^ (temp >>> 32));
219        temp = this.left != +0.0d ? Double.doubleToLongBits(this.left) : 0L;
220        result = 29 * result + (int) (temp ^ (temp >>> 32));
221        temp = this.right != +0.0d ? Double.doubleToLongBits(this.right) : 0L;
222        result = 29 * result + (int) (temp ^ (temp >>> 32));
223        return result;
224    }

237    /**
238     * Creates an adjusted rectangle using the supplied rectangle, the insets
239     * specified by this instance, and the horizontal and vertical
240     * adjustment types.
241     *
242     * @param base  the base rectangle (<code>null</code> not permitted).
243     * @param horizontal  the horizontal adjustment type (<code>null</code> not
244     *                    permitted).
245     * @param vertical  the vertical adjustment type (<code>null</code> not
246     *                  permitted).
247     *
248     * @return The inset rectangle.
249     */
250    public Rectangle2D createAdjustedRectangle(Rectangle2D base,
251            LengthAdjustmentType horizontal, LengthAdjustmentType vertical) {
252        if (base == null) {
253            throw new IllegalArgumentException("Null 'base' argument.");
254        }
255        double x = base.getX();
256        double y = base.getY();
257        double w = base.getWidth();
258        double h = base.getHeight();
259        if (horizontal == LengthAdjustmentType.EXPAND) {
260            double leftOutset = calculateLeftOutset(w);
261            x = x - leftOutset;
262            w = w + leftOutset + calculateRightOutset(w);
263        }
264        else if (horizontal == LengthAdjustmentType.CONTRACT) {
265            double leftMargin = calculateLeftInset(w);
266            x = x + leftMargin;
267            w = w - leftMargin - calculateRightInset(w);
268        }
269        if (vertical == LengthAdjustmentType.EXPAND) {
270            double topMargin = calculateTopOutset(h);
271            y = y - topMargin;
272            h = h + topMargin + calculateBottomOutset(h);
273        }
274        else if (vertical == LengthAdjustmentType.CONTRACT) {
275            double topMargin = calculateTopInset(h);
276            y = y + topMargin;
277            h = h - topMargin - calculateBottomInset(h);
278        }
279        return new Rectangle2D.Double(x, y, w, h);
280    }

293    /**
294     * Creates an 'inset' rectangle.
295     *
296     * @param base  the base rectangle (<code>null</code> not permitted).
297     * @param horizontal  apply horizontal insets?
298     * @param vertical  apply vertical insets?
299     *
300     * @return The inset rectangle.
301     */
302    public Rectangle2D createInsetRectangle(Rectangle2D base,
303            boolean horizontal, boolean vertical) {
304        if (base == null) {
305            throw new IllegalArgumentException("Null 'base' argument.");
306        }
307        double topMargin = 0.0;
308        double bottomMargin = 0.0;
309        if (vertical) {
310            topMargin = calculateTopInset(base.getHeight());
311            bottomMargin = calculateBottomInset(base.getHeight());
312        }
313        double leftMargin = 0.0;
314        double rightMargin = 0.0;
315        if (horizontal) {
316            leftMargin = calculateLeftInset(base.getWidth());
317            rightMargin = calculateRightInset(base.getWidth());
318        }
319        return new Rectangle2D.Double(
320            base.getX() + leftMargin,
321            base.getY() + topMargin,
322            base.getWidth() - leftMargin - rightMargin,
323            base.getHeight() - topMargin - bottomMargin
324        );
325    }

338    /**
339     * Creates an outset rectangle.
340     *
341     * @param base  the base rectangle (<code>null</code> not permitted).
342     * @param horizontal  apply horizontal insets?
343     * @param vertical  apply vertical insets?
344     *
345     * @return An outset rectangle.
346     */
347    public Rectangle2D createOutsetRectangle(Rectangle2D base,
348                                             boolean horizontal,
349                                             boolean vertical) {
350        if (base == null) {
351            throw new IllegalArgumentException("Null 'base' argument.");
352        }
353        double topMargin = 0.0;
354        double bottomMargin = 0.0;
355        if (vertical) {
356            topMargin = calculateTopOutset(base.getHeight());
357            bottomMargin = calculateBottomOutset(base.getHeight());
358        }
359        double leftMargin = 0.0;
360        double rightMargin = 0.0;
361        if (horizontal) {
362            leftMargin = calculateLeftOutset(base.getWidth());
363            rightMargin = calculateRightOutset(base.getWidth());
364        }
365        return new Rectangle2D.Double(
366            base.getX() - leftMargin,
367            base.getY() - topMargin,
368            base.getWidth() + leftMargin + rightMargin,
369            base.getHeight() + topMargin + bottomMargin
370        );
371    }

373    /**
374     * Returns the top margin.
375     *
376     * @param height  the height of the base rectangle.
377     *
378     * @return The top margin (in Java2D units).
379     */
380    public double calculateTopInset(double height) {
381        double result = this.top;
382        if (this.unitType == UnitType.RELATIVE) {
383            result = (this.top * height);
384        }
385        return result;
386    }
387
388    /**
389     * Returns the top margin.
390     *
391     * @param height  the height of the base rectangle.
392     *
393     * @return The top margin (in Java2D units).
394     */
395    public double calculateTopOutset(double height) {
396        double result = this.top;
397        if (this.unitType == UnitType.RELATIVE) {
398            result = (height / (1 - this.top - this.bottom)) * this.top;
399        }
400        return result;
401    }

411        double result = this.bottom;
412        if (this.unitType == UnitType.RELATIVE) {
413            result = (this.bottom * height);

493    /**
494     * Trims the given width to allow for the insets.
495     *
496     * @param width  the width.

500    public double trimWidth(double width) {
501        return width - calculateLeftInset(width) - calculateRightInset(width);

539    /**
540     * Shrinks the given rectangle by the amount of these insets.
541     *
542     * @param area  the area (<code>null</code> not permitted).
543     */
544    public void trim(Rectangle2D area) {
545        double w = area.getWidth();
546        double h = area.getHeight();
547        double l = calculateLeftInset(w);
548        double r = calculateRightInset(w);
549        double t = calculateTopInset(h);
550        double b = calculateBottomInset(h);
551        area.setRect(area.getX() + l, area.getY() + t, w - l - r, h - t - b);
552    }
